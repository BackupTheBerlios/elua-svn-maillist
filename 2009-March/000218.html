+      case 0x43:
+        return KC_RIGHT;
+      case 0x44:
+        return KC_LEFT;               
+    }
+  }
+  else if( data == 0x0D )
+  {
+    // CR/LF sequence, read the second char (LF) if applicable

+      case 0x15:

+      case 0x01:
+        return KC_HOME;

+      case 0x08:
